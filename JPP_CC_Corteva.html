#max and min dont give us much as far as coordinates
xy = cortv['Coordinates']
# dropping null value columns to avoid errors
xy.dropna(inplace = True)
#split string into x and y
newxy = xy.str.split(',', 1, expand=True)
#assign float to xy
xres=[]  
for i in newxy[0]:  
  xres.append(float(i)) 
yres=[]  
for i in newxy[1]:  
  yres.append(float(i)) 
#create data driven map bounds
map_bounds = [[min(yres), min(xres)], 
              [max(yres), max(xres)]]
# Build map 
map_in = folium.Map(bounds=map_bounds, zoom_start=6, 
tiles='cartodbpositron', width=640, height=480)

#only plot unique coordianted instead of all records
for i in range(len(cortv['Coordinates'].unique())):
    
    #give label x and y coords
    lab = str(yres[i])+','+ str(xres[i])  
    #set circle symbol properties
    folium.CircleMarker(
        location=[xres[i], yres[i]],
        #radius=xres[1].value_counts()[i], assign size of marker based on number of occurences #was ugly
        radius=2,
        popup=lab,
        color="#3186cc",
        fill=True,
        fill_color="#3186cc",
        ).add_to(map_in)
